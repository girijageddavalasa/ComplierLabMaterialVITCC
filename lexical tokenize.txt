// c program lexical tokenize

#include <stdio.h>
#include <ctype.h>
#include <string.h>

int isKeyword(char *str) {
    char *keywords[] = {"int","if","while","for","return"};
    for(int i=0;i<5;i++) {
        if(strcmp(str, keywords[i]) == 0)
            return 1;
    }
    return 0;
}

int main() {
    FILE *fp;
    char str[200];

    /* -------- WRITE TO FILE -------- */
    fp = fopen("input.txt", "w");

    printf("Enter code (max 3 lines):\n");
    for(int i=0;i<3;i++) {
        fgets(str, sizeof(str), stdin);
        fputs(str, fp);
    }

    fclose(fp);

    /* -------- READ FROM FILE -------- */
    fp = fopen("input.txt", "r");

    printf("\nTokens:\n");

    while(fgets(str, sizeof(str), fp)) {
        int i = 0;

        while(str[i] != '\0') {

            if(isspace(str[i])) {
                i++;
                continue;
            }

            // Identifier / Keyword
            if(isalpha(str[i]) || str[i]=='_') {
                char buffer[50];
                int j = 0;

                while(isalnum(str[i]) || str[i]=='_') {
                    buffer[j++] = str[i++];
                }
                buffer[j] = '\0';

                if(isKeyword(buffer))
                    printf("KEYWORD : %s\n", buffer);
                else
                    printf("IDENTIFIER : %s\n", buffer);
            }

            // Number
            else if(isdigit(str[i])) {
                char buffer[50];
                int j = 0;

                while(isdigit(str[i]) || str[i]=='.') {
                    buffer[j++] = str[i++];
                }
                buffer[j] = '\0';

                if(strchr(buffer,'.'))
                    printf("FLOAT : %s\n", buffer);
                else
                    printf("INT : %s\n", buffer);
            }

            // ++ and --
            else if(str[i]=='+' && str[i+1]=='+') {
                printf("INC/DEC OPERATOR : ++\n");
                i += 2;
            }
            else if(str[i]=='-' && str[i+1]=='-') {
                printf("INC/DEC OPERATOR : --\n");
                i += 2;
            }

            // Relational
            else if((str[i]=='=' && str[i+1]=='=') ||
                    (str[i]=='!' && str[i+1]=='=') ||
                    (str[i]=='<' && str[i+1]=='=') ||
                    (str[i]=='>' && str[i+1]=='=')) {
                printf("RELATIONAL OPERATOR : %c%c\n", str[i], str[i+1]);
                i += 2;
            }
            else if(str[i]=='<' || str[i]=='>') {
                printf("RELATIONAL OPERATOR : %c\n", str[i]);
                i++;
            }

            // Logical
            else if(str[i]=='&' && str[i+1]=='&') {
                printf("LOGICAL OPERATOR : &&\n");
                i += 2;
            }
            else if(str[i]=='|' && str[i+1]=='|') {
                printf("LOGICAL OPERATOR : ||\n");
                i += 2;
            }
            else if(str[i]=='!') {
                printf("LOGICAL OPERATOR : !\n");
                i++;
            }

            // Assignment
            else if((str[i]=='+'||str[i]=='-'||str[i]=='*'||str[i]=='/') && str[i+1]=='=') {
                printf("ASSIGNMENT OPERATOR : %c=\n", str[i]);
                i += 2;
            }
            else if(str[i]=='=') {
                printf("ASSIGNMENT OPERATOR : =\n");
                i++;
            }

            // Arithmetic
            else if(str[i]=='+' || str[i]=='-' || str[i]=='*' || str[i]=='/') {
                printf("ARITHMETIC OPERATOR : %c\n", str[i]);
                i++;
            }

            // Special symbols
            else if(str[i]==';' || str[i]=='{' || str[i]=='}' || str[i]=='(' || str[i]==')') {
                printf("SPECIAL SYMBOL : %c\n", str[i]);
                i++;
            }

            else {
                printf("UNKNOWN : %c\n", str[i]);
                i++;
            }
        }
    }

    fclose(fp);
    return 0;
}

#include <stdio.h>

/* Function to check keyword */
int isKeyword(char *str) {
    char *keywords[] = {"int","if","while","for","return"};
    for(int i=0;i<5;i++) {
        if(strcmp(str, keywords[i]) == 0)
            return 1;
    }
    return 0;
}

int main() {
    char str[200];

    printf("Enter code:\n");
    fgets(str, sizeof(str), stdin);

    int i = 0;

    while(str[i] != '\0') {

        /* Ignore spaces */
        if(isspace(str[i])) {
            i++;
            continue;
        }

        /* IDENTIFIER / KEYWORD */
        if(isalpha(str[i]) || str[i]=='_') {
            char buffer[50];
            int j = 0;

            while(isalnum(str[i]) || str[i]=='_') {
                buffer[j++] = str[i++];
            }
            buffer[j] = '\0';

            if(isKeyword(buffer))
                printf("KEYWORD : %s\n", buffer);
            else
                printf("IDENTIFIER : %s\n", buffer);
        }

        /* NUMBER */
        else if(isdigit(str[i])) {
            char buffer[50];
            int j = 0;

            while(isdigit(str[i]) || str[i]=='.') {
                buffer[j++] = str[i++];
            }
            buffer[j] = '\0';

            if(strchr(buffer,'.'))
                printf("FLOAT : %s\n", buffer);
            else
                printf("INT : %s\n", buffer);
        }

        /* COMMENT */
        else if(str[i]=='/' && str[i+1]=='/') {
            printf("COMMENT : ");
            while(str[i] != '\0') {
                printf("%c", str[i++]);
            }
        }

        /* INCREMENT / DECREMENT */
        else if(str[i]=='+' && str[i+1]=='+') {
            printf("INC/DEC OPERATOR : ++\n");
            i += 2;
        }

        /* ASSIGNMENT */
        else if(str[i]=='=') {
            printf("ASSIGNMENT OPERATOR : =\n");

        /* ARITHMETIC */

            printf("ARITHMETIC OPERATOR : %c\n", str[i]);
