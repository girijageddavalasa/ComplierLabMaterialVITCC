// odd a even b ccc
%{
#include <stdio.h>
#include <stdlib.h>
%}

%x S2 S3 S4 S5 S6 S7 S8 S9 DEAD

%%

/* -------- S1 (INITIAL) -------- */
<INITIAL>a { BEGIN S2; }
<INITIAL>. { BEGIN DEAD; }

/* -------- S2 -------- */
<S2>a { BEGIN S3; }
<S2>b { BEGIN S4; }
<S2>. { BEGIN DEAD; }

/* -------- S3 -------- */
<S3>a { BEGIN S2; }
<S3>. { BEGIN DEAD; }

/* -------- S4 -------- */
<S4>b { BEGIN S5; }
<S4>. { BEGIN DEAD; }

/* -------- S5 -------- */
<S5>b { BEGIN S6; }
<S5>c { BEGIN S7; }
<S5>. { BEGIN DEAD; }

/* -------- S6 -------- */
<S6>b { BEGIN S5; }
<S6>. { BEGIN DEAD; }

/* -------- S7 -------- */
<S7>c { BEGIN S8; }
<S7>. { BEGIN DEAD; }

/* -------- S8 -------- */
<S8>c { BEGIN S9; }
<S8>. { BEGIN DEAD; }

/* -------- S9 (FINAL) -------- */
<S9>\n {
    printf("\nRESULT: ACCEPTED\n\nEnter string: ");
    BEGIN INITIAL;
}
<S9>. { BEGIN DEAD; }

/* -------- DEAD -------- */
<DEAD>[^\n]+ { }
<DEAD>\n {
    printf("\nRESULT: REJECTED\n\nEnter string: ");
    BEGIN INITIAL;
}

%%

int main() {
    printf("Enter string: ");
    yylex();
    return 0;
}

int yywrap() {
    return 1;
}
