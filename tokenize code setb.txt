/* LEXICAL ANALYZER IN C
   Identifies: keywords, identifiers, constants, comments,
   operators, and special symbols */

#include <stdio.h>
#include <ctype.h>
#include <string.h>

/* -------- FUNCTION TO CHECK KEYWORD -------- */
int isKeyword(char *str) {
    char *keywords[] = {"int","if","while","for","return"};
    int n = 5;

    for(int i = 0; i < n; i++) {
        if(strcmp(str, keywords[i]) == 0)
            return 1;
    }
    return 0;
}

/* -------- MAIN FUNCTION -------- */
int main() {

    char input[500];

    printf("Enter code:\n");

    /* Read full line (can extend to multiple lines if needed) */
    fgets(input, sizeof(input), stdin);

    int i = 0;

    while(input[i] != '\0') {

        /* -------- IGNORE SPACES -------- */
        if(isspace(input[i])) {
            i++;
            continue;
        }

        /* -------- IDENTIFIER / KEYWORD -------- */
        if(isalpha(input[i]) || input[i] == '_') {

            char buffer[50];
            int j = 0;

            while(isalnum(input[i]) || input[i] == '_') {
                buffer[j++] = input[i++];
            }
            buffer[j] = '\0';

            if(isKeyword(buffer))
                printf("KEYWORD : %s\n", buffer);
            else
                printf("IDENTIFIER : %s\n", buffer);
        }

        /* -------- NUMBER (INT / FLOAT) -------- */
        else if(isdigit(input[i])) {

            char buffer[50];
            int j = 0;

            while(isdigit(input[i]) || input[i] == '.') {
                buffer[j++] = input[i++];
            }
            buffer[j] = '\0';

            if(strchr(buffer, '.'))
                printf("FLOAT CONSTANT : %s\n", buffer);
            else
                printf("INTEGER CONSTANT : %s\n", buffer);
        }

        /* -------- COMMENTS -------- */
        else if(input[i] == '/' && input[i+1] == '/') {

            printf("COMMENT : ");

            while(input[i] != '\0') {
                printf("%c", input[i++]);
            }
        }

        else if(input[i] == '/' && input[i+1] == '*') {

            printf("COMMENT : ");

            i += 2;

            while(!(input[i] == '*' && input[i+1] == '/') && input[i] != '\0') {
                printf("%c", input[i++]);
            }

            i += 2; /* skip closing */
            printf("\n");
        }

        /* -------- INCREMENT / DECREMENT -------- */
        else if(input[i] == '+' && input[i+1] == '+') {
            printf("INCREMENT OPERATOR : ++\n");
            i += 2;
        }
        else if(input[i] == '-' && input[i+1] == '-') {
            printf("DECREMENT OPERATOR : --\n");
            i += 2;
        }

        /* -------- RELATIONAL OPERATORS -------- */
        else if((input[i]=='=' && input[i+1]=='=') ||
                (input[i]=='!' && input[i+1]=='=') ||
                (input[i]=='<' && input[i+1]=='=') ||
                (input[i]=='>' && input[i+1]=='=')) {

            printf("RELATIONAL OPERATOR : %c%c\n", input[i], input[i+1]);
            i += 2;
        }
        else if(input[i]=='<' || input[i]=='>' ) {
            printf("RELATIONAL OPERATOR : %c\n", input[i]);
            i++;
        }

        /* -------- LOGICAL OPERATORS -------- */
        else if(input[i]=='&' && input[i+1]=='&') {
            printf("LOGICAL OPERATOR : &&\n");
            i += 2;
        }
        else if(input[i]=='|' && input[i+1]=='|') {
            printf("LOGICAL OPERATOR : ||\n");
            i += 2;
        }
        else if(input[i]=='!') {
            printf("LOGICAL OPERATOR : !\n");
            i++;
        }

        /* -------- ASSIGNMENT OPERATORS -------- */
        else if((input[i]=='+' || input[i]=='-' || input[i]=='*' || input[i]=='/')
                && input[i+1]=='=') {

            printf("ASSIGNMENT OPERATOR : %c=\n", input[i]);
            i += 2;
        }
        else if(input[i]=='=') {
            printf("ASSIGNMENT OPERATOR : =\n");
            i++;
        }

        /* -------- ARITHMETIC OPERATORS -------- */
        else if(input[i]=='+' || input[i]=='-' || input[i]=='*' || input[i]=='/') {
            printf("ARITHMETIC OPERATOR : %c\n", input[i]);
            i++;
        }

        /* -------- SPECIAL SYMBOLS -------- */
        else if(input[i]==';' || input[i]=='{' || input[i]=='}' ||
                input[i]=='(' || input[i]==')') {

            printf("SPECIAL SYMBOL : %c\n", input[i]);
            i++;
        }

        /* -------- UNKNOWN -------- */
        else {
            printf("UNKNOWN : %c\n", input[i]);
            i++;
        }
    }

    return 0;
}

        }

        /* ASSIGNMENT */
        else if((input[i]=='+' || input[i]=='-' || input[i]=='*' || input[i]=='/')
                && input[i+1]=='=') {

            printf("ASSIGNMENT OPERATOR : %c=\n", input[i]);
            i += 2;
        }
        else if(input[i]=='=') {
            printf("ASSIGNMENT OPERATOR : =\n");
            i++;
        }

        /* ARITHMETIC */
        else if(input[i]=='+' || input[i]=='-' || input[i]=='*' || input[i]=='/') {
            printf("ARITHMETIC OPERATOR : %c\n", input[i]);
            i++;
        }

        /* SYMBOLS */
        else if(input[i]==';' || input[i]=='{' || input[i]=='}' ||
                input[i]=='(' || input[i]==')') {

            printf("SPECIAL SYMBOL : %c\n", input[i]);
            i++;
        }

        else {
            printf("UNKNOWN : %c\n", input[i]);
            i++;
        }
    }
}

/* -------- MAIN -------- */
int main() {

    char input[1000] = "";
    char line[200];

    printf("Enter code (type END to stop):\n");

    while(1) {
        fgets(line, sizeof(line), stdin);

        /* remove newline */
        line[strcspn(line, "\n")] = 0;

        /* stop condition */
        if(strcmp(line, "END") == 0)
            break;

        strcat(input, line);
        strcat(input, "\n");
    }

    printf("\n--- TOKENS ---\n");

    analyze(input);

    return 0;
}
