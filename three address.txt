%{
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int temp = 0;

char* newtemp() {
    static char t[10];
    sprintf(t, "t%d", temp++);   // create t0, t1...
    return t;
}

void yyerror(char *s) {
    printf("Error\n");
}

int yylex();
%}

%union {
    int ival;
    char sval[20];
}

%token <ival> NUM
%token <sval> ID

%type <sval> expr term power factor

%%

stmt : ID '=' expr ';'
      { printf("%s = %s\n", $1, $3); }   // final result
     ;

/* + and - (lowest priority) */
expr : expr '+' term {
        char *t = newtemp();
        printf("%s = %s + %s\n", t, $1, $3);
        strcpy($$, t);
     }
     | expr '-' term {
        char *t = newtemp();
        printf("%s = %s - %s\n", t, $1, $3);
        strcpy($$, t);
     }
     | term { strcpy($$, $1); }
     ;

/* * and / */
term : term '*' power {
        char *t = newtemp();
        printf("%s = %s * %s\n", t, $1, $3);
        strcpy($$, t);
     }
     | term '/' power {
        char *t = newtemp();
        printf("%s = %s / %s\n", t, $1, $3);
        strcpy($$, t);
     }
     | power { strcpy($$, $1); }
     ;

/* ^ (highest priority) */
power : factor '^' power {
        char *t = newtemp();
        printf("%s = %s ^ %s\n", t, $1, $3);
        strcpy($$, t);
     }
     | factor { strcpy($$, $1); }
     ;

factor : '(' expr ')' { strcpy($$, $2); }
       | NUM { sprintf($$, "%d", $1); }
       | ID  { strcpy($$, $1); }
       ;

%%

int main() {
    printf("Enter: ");
    yyparse();
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

char input[100];
int pos = 0;
int tempCount = 0;

/* Generate temp variables t0, t1... */
char* newtemp() {
    static char temp[10];
    sprintf(temp, "t%d", tempCount++);
    return temp;
}

/* Skip spaces */
void skipSpaces() {
    while (isspace(input[pos])) pos++;
}

/* Read identifier */
char* parseID() {
    static char id[20];
    int i = 0;

    skipSpaces();

    while (isalnum(input[pos])) {
        id[i++] = input[pos++];
    }
    id[i] = '\0';

    return id;
}

/* Read number */
char* parseNUM() {
    static char num[20];
    int i = 0;

    skipSpaces();

    while (isdigit(input[pos])) {
        num[i++] = input[pos++];
    }
    num[i] = '\0';

    return num;
}

/* Forward declarations */
char* expr();
char* term();
char* power();
char* factor();

/* factor -> (expr) | ID | NUM */
char* factor() {
    skipSpaces();

    if (input[pos] == '(') {
        pos++;
        char* e = expr();
        skipSpaces();
        if (input[pos] == ')') pos++;
        return e;
    }

    if (isalpha(input[pos])) {
        return parseID();
    }

    if (isdigit(input[pos])) {
        return parseNUM();
    }

    return "";
}

/* power -> factor ^ power */
char* power() {
    char* left = factor();
    skipSpaces();

    if (input[pos] == '^') {
        pos++;
        char* right = power();

        char* t = newtemp();
        printf("%s = %s ^ %s\n", t, left, right);
        return t;
    }

    return left;
}

/* term -> term * power | term / power */
char* term() {
    char* left = power();

    while (1) {
        skipSpaces();

        if (input[pos] == '*') {
            pos++;
            char* right = power();

            char* t = newtemp();
            printf("%s = %s * %s\n", t, left, right);
            left = t;
        }
        else if (input[pos] == '/') {
            pos++;
            char* right = power();

            char* t = newtemp();
            printf("%s = %s / %s\n", t, left, right);
            left = t;
        }
        else break;
    }

    return left;
}

/* expr -> expr + term | expr - term */
char* expr() {
    char* left = term();

    while (1) {
        skipSpaces();

        if (input[pos] == '+') {
            pos++;
            char* right = term();

            char* t = newtemp();
            printf("%s = %s + %s\n", t, left, right);
            left = t;
        }
        else if (input[pos] == '-') {
            pos++;
            char* right = term();

            char* t = newtemp();
            printf("%s = %s - %s\n", t, left, right);
            left = t;
        }
        else break;
    }

    return left;
}

/* stmt -> ID = expr ; */
void parseStatement() {
    char lhs[20];

    skipSpaces();
    strcpy(lhs, parseID());

    skipSpaces();
    if (input[pos] != '=') {
        printf("Error: Expected '='\n");
        return;
    }
    pos++;

    char* result = expr();

    skipSpaces();
    if (input[pos] != ';') {
        printf("Error: Expected ';'\n");
        return;
    }

    printf("%s = %s\n", lhs, result);
}

/* MAIN */
int main() {
    printf("Enter expression: ");
    fgets(input, sizeof(input), stdin);

    parseStatement();

    return 0;
}
