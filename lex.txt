// c code

#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX_WORDS 1000
#define MAX_LEN 50

int main() {
    FILE *fp;
    char word[MAX_LEN];
    char words[MAX_WORDS][MAX_LEN];
    int count[MAX_WORDS] = {0};
    int i, j, found, total = 0;

    fp = fopen("input.txt", "r");
    if (fp == NULL) {
        printf("Error opening file!\n");
        return 1;
    }

    while (fscanf(fp, "%s", word) != EOF) {
        // Convert word to lowercase
        for (i = 0; word[i]; i++) {
            word[i] = tolower(word[i]);
        }

        // Remove punctuation (basic)
        int len = strlen(word);
        while (len > 0 && ispunct(word[len - 1])) {
            word[len - 1] = '\0';
            len--;
        }

        found = 0;

        // Check if word already exists
        for (i = 0; i < total; i++) {
            if (strcmp(words[i], word) == 0) {
                count[i]++;
                found = 1;
                break;
            }
        }

        // If new word
        if (!found) {
            strcpy(words[total], word);
            count[total] = 1;
            total++;
        }
    }

    fclose(fp);

    // Print results
    printf("\nWord Frequencies:\n");
    for (i = 0; i < total; i++) {
        printf("%s : %d\n", words[i], count[i]);
    }

    return 0;
}

#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX_WORDS 1000
#define MAX_LEN 50
#define MAX_LINE 200

int main() {
    char line[MAX_LINE];
    char word[MAX_LEN];
    char words[MAX_WORDS][MAX_LEN];
    int count[MAX_WORDS] = {0};

    int i, j, k, found, total = 0;

    printf("Enter text (type END to stop):\n");

    while (1) {
        fgets(line, sizeof(line), stdin);

        // Stop condition
        if (strncmp(line, "END", 3) == 0)
            break;

        i = 0;

        while (line[i] != '\0') {

            // Skip spaces
            while (line[i] == ' ' || line[i] == '\t' || line[i] == '\n')
                i++;

            j = 0;

            // Read a word
            while (line[i] != '\0' && !isspace(line[i]) && !ispunct(line[i])) {
                word[j++] = tolower(line[i]);
                i++;
            }

            word[j] = '\0';

            if (j == 0) {
                i++;
                continue;
            }

            found = 0;

            // Check if word already exists
            for (k = 0; k < total; k++) {
                if (strcmp(words[k], word) == 0) {
                    count[k]++;
                    found = 1;
                    break;
                }
            }

            // New word
            if (!found) {
                strcpy(words[total], word);
                count[total] = 1;
                total++;
            }
        }
    }

    // Print results
    printf("\nWord Frequencies:\n");
    for (i = 0; i < total; i++) {
        printf("%s : %d\n", words[i], count[i]);
    }

    return 0;
}

// one only  count
#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX_WORDS 1000
#define MAX_LEN 50
#define MAX_LINE 200

int main() {
    char line[MAX_LINE];
    char word[MAX_LEN];
    char words[MAX_WORDS][MAX_LEN];
    int count[MAX_WORDS] = {0};

    int i, j, k, found, total = 0;

    printf("Enter text (type END to stop):\n");

    while (1) {
        fgets(line, sizeof(line), stdin);

        if (strncmp(line, "END", 3) == 0)
            break;

        i = 0;

        while (line[i] != '\0') {

            // Skip spaces
            while (isspace(line[i]))
                i++;

            j = 0;

            // Extract word
            while (line[i] != '\0' && !isspace(line[i]) && !ispunct(line[i])) {
                word[j++] = tolower(line[i]);
                i++;
            }

            word[j] = '\0';

            if (j == 0) {
                i++;
                continue;
            }

            found = -1;

            // Check if word exists
            for (k = 0; k < total; k++) {
                if (strcmp(words[k], word) == 0) {
                    found = k;
                    break;
                }
            }

            if (found != -1) {
                count[found]++;
            } else {
                strcpy(words[total], word);
                count[total] = 1;
                total++;
            }
        }
    }

    // Find max frequency (latest wins in tie)
    int max = -1;
    int index = -1;

    for (i = 0; i < total; i++) {
        if (count[i] >= max) {  // >= ensures latest wins
            max = count[i];
            index = i;
        }
    }

    // Output result
    if (index != -1) {
        printf("\nMost frequent word: %s\n", words[index]);
        printf("Count: %d\n", max);
    }

    return 0;
}

// longest words

%{
#include <stdio.h>
#include <string.h>

char top[3][100];   // store top 3 words
int len[3] = {0,0,0};
%}

%%

[a-zA-Z]+ {
    int l = strlen(yytext);

    // Check if this word is in top 3
    for(int i=0;i<3;i++) {
        if(l >= len[i]) {

            // shift down
            for(int j=2;j>i;j--) {
                len[j] = len[j-1];
                strcpy(top[j], top[j-1]);
            }

            // insert new word
            len[i] = l;
            strcpy(top[i], yytext);
            break;
        }
    }
}

[ \t\n]+ ;
. ;

%%

int main() {
    FILE *fp;

    fp = fopen("input.txt", "r");
    yyin = fp;

    yylex();

    printf("\nLongest word: %s (length = %d)\n", top[0], len[0]);

    printf("\nTop 3 longest words:\n");
    for(int i=0;i<3;i++) {
        if(len[i] > 0)
            printf("%s (length = %d)\n", top[i], len[i]);
    }

    fclose(fp);
    return 0;
}
