// .y code 
%{
#include <stdio.h>
#include <stdlib.h>

extern int a_count;
extern int b_count;

void yyerror(const char *msg);
int yylex();
%}

%token TOK_A TOK_B NEWLINE INVALID

%%

program:
    | program line
    ;

line:
    NEWLINE
    | string NEWLINE {
        if (a_count != b_count) {
            printf("String ACCEPTED  (a^%d b^%d, %d != %d)\n",
                   a_count, b_count, a_count, b_count);
        } else {
            printf("String REJECTED  (a^%d b^%d, counts are equal)\n",
                   a_count, b_count);
        }
        a_count = 0;
        b_count = 0;
    }
    | INVALID NEWLINE {
        printf("String REJECTED  (invalid character)\n");
        a_count = 0;
        b_count = 0;
    }
    ;

string:
    a_part b_part
    ;

a_part:
    TOK_A
    | a_part TOK_A
    ;

b_part:
    TOK_B
    | b_part TOK_B
    ;

%%

void yyerror(const char *msg) {
    printf("String REJECTED  (parse error)\n");
    a_count = 0;
    b_count = 0;
}

int main() {
    printf("L(G) = { a^n b^m | m != n }\n");
    printf("Enter string:\n");
    yyparse();
    return 0;
}

%{

#include <stdlib.h>

void yyerror(const char *msg);
int  yylex();

int accepted = 0;
%}

%token TOK_A TOK_B NEWLINE INVALID

%%

program:
    | program line
    ;

line:
    NEWLINE
    | string NEWLINE {
        if (accepted)
            printf("String ACCEPTED  => belongs to L(G)\n");
        else
            printf("String REJECTED  => NOT in L(G)\n");
        accepted = 0;
    }
    | INVALID NEWLINE {
        printf("String REJECTED  (invalid character)\n");
    }
    ;

string:
    TOK_A TOK_B mid { accepted = 1; }
    ;

mid:
    TOK_B TOK_B TOK_A TOK_A mid TOK_B TOK_A
    | TOK_B TOK_B TOK_A
    ;

%%

void yyerror(const char *msg) {
    printf("String REJECTED  (does not match pattern: %s)\n", msg);
    accepted = 0;
}

int main() {
    printf("=================================================================\n");
    printf("  Experiment 15 - Language 2                                     \n");
    printf("  L(G) = { ab(bbaa)^n bba(ba)^n | n >= 0 }                      \n");
    printf("=================================================================\n");
    printf("Pattern:  ab  [bbaa]^n  bba  [ba]^n\n\n");
    printf("Expected strings:\n");
    printf("  n=0 :  ab bba               =>  abbba\n");
    printf("  n=1 :  ab bbaa bba ba       =>  abbbaabbaba\n");
    printf("  n=2 :  ab bbaa bbaa bba ba ba\n\n");
    printf("Enter a string and press Enter. Press Ctrl+D to quit.\n\n");
    yyparse();
    return 0;
}
